/** * 教習所 商圏人口一括算出スクリプト（GAS版） * 機能： * 1. 住所 → 緯度経度（Google Geocoding） * 2. 緯度経度 → 周辺500mメッシュの15〜19歳人口取得（e-Stat API） * 3. 15分圏（12.5km）・25分圏（20.8km）内の人口集計 * 4. ÷5 で免許取得見込人口を算出 */ const E_STAT_APP_ID = '35c0f09bdf99c574e265036e1719e89b0a81536c'; const CENSUS_STATS_CODE = '00200521'; const CENSUS_SURVEY_YEAR = '2020'; const SPEED_KM = 50; const RADIUS_15MIN_KM = SPEED_KM * 15 / 60; const RADIUS_25MIN_KM = SPEED_KM * 25 / 60; const SHEET_SOURCE = 'ジオコーディング対象'; const SHEET_RESULT = '商圏人口結果'; function onOpen() { const ui = SpreadsheetApp.getUi(); ui.createMenu('教習所分析') .addItem('① ジオコーディング実行', 'step1_geocode') .addItem('② 商圏人口を算出', 'step2_calcPopulation') .addItem('③ 結果シートを作成', 'step3_createResultSheet') .addSeparator() .addItem('全ステップ一括実行', 'runAll') .addToUi(); } function runAll() { step1_geocode(); step2_calcPopulation(); step3_createResultSheet(); } function step1_geocode() { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_SOURCE); if (geo.status === 'OK' && geo.results.length > 0) { const loc = geo.results[0].geometry.location; sheet.getRange(i, 6).setValue(loc.lat); sheet.getRange(i, 7).setValue(loc.lng); sheet.getRange(i, 8).setValue('取得済'); Logger.log('Geocode: ' + i + '/' + lastRow); } } SpreadsheetApp.flush(); SpreadsheetApp.getUi().alert('ジオコーディング完了\n成功: ' + success + ' / 失敗: ' + fail + ' / スキップ: ' + skip); } function step2_calcPopulation() { if (!E_STAT_APP_ID || E_STAT_APP_ID.length < 10) { SpreadsheetApp.getUi().alert('e-StatのアプリケーションIDを設定してください'); return; } const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_SOURCE); if (!sheet) { SpreadsheetApp.getUi().alert(SHEET_SOURCE + 'シートが見つかりません'); return; } const lastRow = sheet.getLastRow(); const headers = ['15分圏_15-19歳', '25分圏_15-19歳', '15分圏_見込(÷5)', '25分圏_見込(÷5)', '15分圏_16-18歳(×0.6)', '25分圏_16-18歳(×0.6)']; for (let j = 0; j < headers.length; j++) { sheet.getRange(1, 10 + j).setValue(headers[j]); } const prefCodes = getPrefCodeMap(); const schoolsByPref = {}; for (let i = 2; i <= lastRow; i++) { const lat = sheet.getRange(i, 6).getValue(); const lng = sheet.getRange(i, 7).getValue(); if (!lat || !lng) continue; const pref = String(sheet.getRange(i, 2).getValue()); if (!schoolsByPref[pref]) schoolsByPref[pref] = []; schoolsByPref[pref].push({ row: i, lat: Number(lat), lng: Number(lng) }); } let processed = 0; const totalPrefs = Object.keys(schoolsByPref).length; for (const pref of Object.keys(schoolsByPref)) { processed++; Logger.log('処理中: ' + pref + ' (' + processed + '/' + totalPrefs + ')'); const prefCode = prefCodes[pref]; if (!prefCode) { Logger.log('都道府県コード不明: ' + pref); continue; } try { const tableIds = findStatsTableIds(prefCode); if (!tableIds || tableIds.length === 0) { Logger.log(pref + ': 統計表ID取得失敗'); continue; } const areaPopData = getPopulationFromTables(tableIds, prefCode); if (!areaPopData || areaPopData.length === 0) { Logger.log(pref + ': 人口データ取得失敗'); continue; } Logger.log(pref + ': ' + areaPopData.length + '件の小地域データ取得'); } else if (dist <= RADIUS_25MIN_KM) { pop25min += area.pop15_19 || 0; } } pop25min += pop15min; sheet.getRange(school.row, 10).setValue(pop15min); sheet.getRange(school.row, 11).setValue(pop25min); sheet.getRange(school.row, 12).setValue(Math.round(pop15min / 5)); sheet.getRange(school.row, 13).setValue(Math.round(pop25min / 5)); sheet.getRange(school.row, 14).setValue(Math.round(pop15min * 0.6)); sheet.getRange(school.row, 15).setValue(Math.round(pop25min * 0.6)); } } catch (e) { Logger.log(pref + ' エラー: ' + e.message); } SpreadsheetApp.flush(); Utilities.sleep(1000); } SpreadsheetApp.getUi().alert('商圏人口の算出が完了しました'); } function step3_createResultSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const src = ss.getSheetByName(SHEET_SOURCE); if (!src) { SpreadsheetApp.getUi().alert(SHEET_SOURCE + 'シートが見つかりません'); return; } let dst = ss.getSheetByName(SHEET_RESULT); if (dst) ss.deleteSheet(dst); dst = ss.insertSheet(SHEET_RESULT); const headers = ['教習所名','都道府県','ID','2023年卒業生数','緯度','経度','15分圏_15-19歳','25分圏_15-19歳','15分圏_見込(÷5)','25分圏_見込(÷5)','15分圏_16-18歳(×0.6)','25分圏_16-18歳(×0.6)','15分圏シェア(%)','25分圏シェア(%)']; dst.getRange(1, 1, 1, headers.length).setValues([headers]); dst.getRange(1, 1, 1, headers.length).setFontWeight('bold'); const lastRow = src.getLastRow(); let outRow = 2; for (let i = 2; i <= lastRow; i++) { const lat = src.getRange(i, 6).getValue(); if (!lat) continue; const name = src.getRange(i, 4).getValue(); const pref = src.getRange(i, 2).getValue(); const id = src.getRange(i, 3).getValue(); Math.round(grad / c25 * 1000) / 10 : ''; dst.getRange(outRow, 1, 1, headers.length).setValues([[name, pref, id, grad, lat, lng, p15, p25, d15, d25, c15, c25, s15, s25]]); outRow++; } dst.autoResizeColumns(1, headers.length); SpreadsheetApp.getUi().alert('結果シート作成完了（' + (outRow - 2) + '件）'); } // ===== e-Stat API ===== function findStatsTableIds(prefCode) { const url = 'https://api.e-stat.go.jp/rest/3.0/app/json/getStatsList' + '?appId=' + E_STAT_APP_ID + '&surveyYears=' + CENSUS_SURVEY_YEAR + '&statsCode=' + CENSUS_STATS_CODE + '&searchKind=2' + '&searchWord=年齢'; try { const response = UrlFetchApp.fetch(url, { muteHttpExceptions: true }); const json = JSON.parse(response.getContentText()); const tables = json.GET_STATS_LIST.DATALIST_INF.TABLE_INF; if (!tables) return []; const arr = Array.isArray(tables) ? tables : [tables]; const ids = []; for (const t of arr) { const title = (t.TITLE && t.TITLE['$']) ? t.TITLE['$'] : String(t.TITLE || ''); if (title.includes('年齢') && title.includes('5歳階級') && title.includes('男女別人口')) { ids.push(t['@id']); } } Logger.log('統計表ID候補: ' + ids.length + '件'); return ids; } catch (e) { Logger.log('findStatsTableIds error: ' + e.message); return []; } } function getPopulationFromTables(tableIds, prefCode) { const results = []; for (const tableId of tableIds) { try { const data = fetchPopData(tableId, prefCode); if (data.length > 0) { Logger.log('tableId ' + tableId + ': ' + data.length + '件取得'); return data; } } catch (e) { Logger.log('tableId ' + tableId + ' error: ' + e.message); } Utilities.sleep(1000); } return results; } function fetchPopData(tableId, prefCode) { const results = []; let startPos = 1; const limit = 100000; // まずメタ情報を取得して15-19歳のカテゴリコードを特定 const metaUrl = 'https://api.e-stat.go.jp/rest/3.0/app/json/getMetaInfo' + '?appId=' + E_STAT_APP_ID + '&statsDataId=' + tableId; for (const cls of (Array.isArray(classObj) ? classObj : [classObj])) { if (cls['@id'] === 'cat01') { const cats = Array.isArray(cls.CLASS) ? cls.CLASS : [cls.CLASS]; for (const c of cats) { const name = c['@name'] || ''; if (name.includes('15～19') || name.includes('15〜19') || name === '15～19歳') { cat01Code_15_19 = c['@code']; Logger.log('15-19歳コード: ' + cat01Code_15_19 + ' (' + name + ')'); break; } } } } } catch (e) { Logger.log('メタ情報取得エラー: ' + e.message); } if (!cat01Code_15_19) { Logger.log('15-19歳のカテゴリコードが見つかりません'); return []; } // データ取得（都道府県フィルタ + 15-19歳フィルタ + 男女総数） const url = 'https://api.e-stat.go.jp/rest/3.0/app/json/getStatsData' + '?appId=' + E_STAT_APP_ID + '&statsDataId=' + tableId + '&cdCat01=' + cat01Code_15_19 + '&cdArea=' + prefCode + '&startPosition=' + startPos + '&limit=' + limit; try { const response = UrlFetchApp.fetch(url, { muteHttpExceptions: true }); const json = JSON.parse(response.getContentText()); if (json.GET_STATS_DATA.RESULT.STATUS !== 0) { Logger.log('API Error: ' + JSON.stringify(json.GET_STATS_DATA.RESULT)); return []; } const data = json.GET_STATS_DATA.STATISTICAL_DATA; if (!data || !data.DATA_INF || !data.DATA_INF.VALUE) return []; const values = Array.isArray(data.DATA_INF.VALUE) ? data.DATA_INF.VALUE : [data.DATA_INF.VALUE]; // 地域コード→座標のマッピングが必要 // e-Stat小地域には座標がないため、市区町村の代表座標で代用 const areaCoordsMap = getMunicipalityCoords(prefCode); // 地域コードごとに人口を集計 const areaMap = {}; for (const v of values) { const areaCode = v['@area']; i <= 999; i++) { const munCode = prefCode + String(i).padStart(3, '0'); coordsMap[munCode] = { lat: loc.lat, lng: loc.lng }; } } } catch (e) { Logger.log('getMunicipalityCoords error: ' + e.message); } return coordsMap; } // ===== ユーティリティ ===== function haversine(lat1, lng1, lat2, lng2) { const R = 6371; const dLat = (lat2 - lat1) * Math.PI / 180; const dLng = (lng2 - lng1) * Math.PI / 180; const a = Math.sin(dLat / 2) * Math.sin(dLat / 2) + Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) * Math.sin(dLng / 2) * Math.sin(dLng / 2);
